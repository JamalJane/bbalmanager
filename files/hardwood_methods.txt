1. SUPABASE SETUP & CONFIGURATION
================================================================================

1.1 Client Initialization
--------------------------
One client file imported everywhere. Store keys in .env — never commit them.

  // src/lib/supabase.js
  import { createClient } from '@supabase/supabase-js'
  const supabaseUrl = import.meta.env.VITE_SUPABASE_URL
  const supabaseKey = import.meta.env.VITE_SUPABASE_ANON_KEY
  export const supabase = createClient(supabaseUrl, supabaseKey)

NOTE: Use VITE_ prefix for all env vars in Vite. Never expose service_role key.

1.2 Environment Variables
--------------------------
  VITE_SUPABASE_URL       Supabase Dashboard > Settings > API > Project URL
  VITE_SUPABASE_ANON_KEY  Supabase Dashboard > Settings > API > Anon key

1.3 RLS Policy
--------------
All tables have RLS enabled. For single-player, allow all for authenticated:

  create policy "authenticated full access" on players
    for all using (auth.role() = 'authenticated');

Apply same policy to every table. For read-only lookup tables (persona
definitions, pathway definitions, templates, etc.) use public read:

  create policy "public read" on persona_definitions for select using (true);

1.4 RPC Functions Needed
-------------------------
  increment_legacy_score(gm_id, delta)
    Atomically adds delta to gm_profiles.legacy_score. Floors at 0.

  increment_relationship(gm_id, ai_team_id, delta)
    Updates relationship score between player and AI team.

  advance_season_week(season_id)
    Increments games_played, checks if season complete.

  calc_draft_order(draft_class_id)
    Reads final standings, generates pick slots, inserts to draft_picks.

  resolve_narrative_event(event_id, choice)
    Marks event resolved, records chosen option, applies effects.

  export function useXxx(param) {
    const [data, setData]       = useState(null)
    const [loading, setLoading] = useState(true)
    const [error, setError]     = useState(null)

    const fetch = useCallback(async () => {
      setLoading(true)

        .from('table').select('*').eq('col', param)

      else setData(data)

    useEffect(() => { if (param) fetch() }, [param, fetch])
    return { data, loading, error, refetch: fetch }
  }

4. SIMENGINE.JS — HOW WINNING & LOSING WORKS
================================================================================

4.1 The Core Concept
---------------------
No actual gameplay. Winning and losing is a weighted dice roll. Your management
decisions over an entire season determine your team's strength number. The sim
reflects that. The game is won in the front office.

4.2 Team Strength Formula
--------------------------
  const CHEMISTRY_MULT = { high: 1.03, mid: 1.0, low: 0.97 }
  const HOME_COURT     = 3

  function calcTeamStrength(team, isHome = false) {
    // 1. Starting five average overall
    const starterAvg = team.starters.reduce((s, p) => s + p.overall, 0) / 5

    // 2. Chemistry multiplier
    const chemMult = team.chemistry >= 70 ? CHEMISTRY_MULT.high
                   : team.chemistry < 40  ? CHEMISTRY_MULT.low
                   : CHEMISTRY_MULT.mid

    // 3. Coaching bonus (best coach level / 5 * 2, max +2)
    const bestCoachLevel = Math.max(...team.coaches.map(c => c.level), 1)
    const coachBonus = (bestCoachLevel / 5) * 2

    // 4. Home court
    const homeBonus = isHome ? HOME_COURT : 0

    return (starterAvg * chemMult) + coachBonus + homeBonus
  }

4.3 Variance & Score Generation
---------------------------------
  function simGame(homeTeam, awayTeam) {
    const homeStr = calcTeamStrength(homeTeam, true)
    const awayStr = calcTeamStrength(awayTeam, false)

    // Variance: +-8 applied independently
    const homeVariance = (Math.random() * 16) - 8
    const awayVariance = (Math.random() * 16) - 8

    // Map to realistic score range 85-115
    const homeScore = Math.round(85 + (homeStr - 65) + homeVariance)
    const awayScore = Math.round(85 + (awayStr - 65) + awayVariance)

    return {
      homeScore: Math.max(70, homeScore),
      awayScore: Math.max(70, awayScore),
      winner: homeScore > awayScore ? homeTeam.id : awayTeam.id
    }
  }

NOTE: +-8 variance means a 5-point weaker team wins ~25% of the time.

4.4 How Dev Decisions Feed Strength
-------------------------------------
  Correct pathway assigned     → Player overall grows faster
  Persona-pathway synergy      → 1.5x growth multiplier
  High team chemistry          → 1.03x strength multiplier every game
  Upgraded coaching staff      → Coach bonus grows from +0.4 to +2
  Called up ready dev player   → Higher overall in starting five
  Signed high-overall FA       → Immediately raises starter avg
  Cancer + Cornerstone clash   → Chemistry drops, 0.97x penalty applies
  Wrong pathway for persona    → 0.55x growth, player stagnates

4.5 Key Moment Generation
--------------------------
  function generateKeyMoments(homeTeam, awayTeam, homeScore, awayScore) {
    const moments = []
    const scoreDiff = Math.abs(homeScore - awayScore)
    const allPlayers = [...homeTeam.starters, ...awayTeam.starters]

    // 1. Momentum swing — 40% chance any game
    if (Math.random() < 0.4)
      moments.push({ event_type: 'momentum_swing', ... })

    // 2. Clutch time — if final score within 5
    if (scoreDiff <= 5)
      moments.push({ event_type: 'clutch_moment', is_key_moment: true, ... })

    // 3. Persona moments — Clutch/IceInVeins trait elevated in close game
    allPlayers.forEach(p => {
      const hasClutchTrait = p.trait_tags?.includes('Ice in his Veins')
                          || p.trait_tags?.includes('Clutch')
      const prob = (hasClutchTrait && scoreDiff <= 5) ? 0.7 : 0.15
      if (Math.random() < prob)
        moments.push({ event_type: 'persona_moment', player_id: p.id, ... })
    })

    // 4. Injury scare — 2% per starter
    allPlayers.forEach(p => {
      if (Math.random() < 0.02)
        moments.push({ event_type: 'injury_scare', player_id: p.id, ... })
    })

    // 5. Breakout alert — if player exceeds 1.4x season avg
    allPlayers.forEach(p => {
      if (p.overall > p.season_avg_overall * 1.4 && Math.random() < 0.5)
        moments.push({ event_type: 'breakout_alert', player_id: p.id, ... })
    })

    return moments
  }

5. OTHER ENGINE METHODS
================================================================================

5.1 devEngine.js
-----------------
growthRoll(player, rosterEntry, coaches) → { attribute, delta } | null
  Weekly dev tick. Returns attribute to improve and delta (+1, +2, -1) or null.
  Inputs: age multiplier (peaks 22-24), potential/100, minutes multiplier,
          coach boost for pathway specialty, persona-pathway compat multiplier.

  Age multipliers: 18=1.2 19=1.3 20=1.4 21=1.45 22-23=1.5 24=1.45
                   25=1.35 26=1.25 27=1.15 28=1.05 29=0.95 30=0.85 31+=0.75

  Minutes multipliers: 28+ mins=1.2, 18-27=1.0, under 18=0.8

  Probability: roll < baseChance*0.05 = breakout (+2)
               roll < baseChance*0.40 = normal (+1)
               roll < baseChance*0.42 = regression (-1)
               else = no change

getCompatMult(personaSub, pathway) → number
  Queries persona_pathway_compatibility table. Returns 0.55-1.5.
  Default 1.0 if no row found.

checkPersonaEvolution(player, seasonEvents) → newPersona | null
  Season end check. Evaluates evolution_trigger from persona_definitions.

checkTraitTagUnlock(player, pathway) → traitTag | null
  Checks if player hit attribute threshold for pathway trait tag.

getReadinessScore(player) → number (0-100)
  Dev league call-up readiness. Score >= 75 triggers story beat.
  Formula: (overall-60)*2 + potential*0.3 + morale*0.2 + age bonus

5.2 tradeEngine.js
-------------------
calculatePlayerValue(player) → number
  (overall x ageMult) + (contract_years x salaryEfficiency) + contract bonus
  Contract bonus: 3+ years = +15, 1 year = -20

  PICK_VALUES: top5=75, lottery=70, mid=65, late=58, comp=50

evaluateOffer(offer, aiTeam) → { decision, gap }
  Compares offer value to ask value. Returns accept/counter/reject.
  Personality thresholds shift what gap is acceptable:
    shark:    hardest, accept threshold -0.05
    builder:  flexible on non-youth, accept threshold +0.05
    closer:   overpays for vets, accept threshold +0.08
    analyst:  purely fair, accept threshold +0.10
    loyalist: slow but fair, accept threshold +0.05

generateCounter(offer, aiTeam, round) → counterOffer | null
  Round 1: asks for more from you.
  Round 2: offers more from them.
  Round 3: final — collapse if still no deal.
  Never moves both needles in same counter.

5.3 storyBeats.js
------------------
generateStoryBeat(eventType, data) → { event_type, description }
  Picks random template from pool for eventType.
  Replaces {variable} placeholders with real data values.

  12 event types: pathway_switch, breakout_game, persona_conflict,
  position_shift, controversial_trade, dev_league_ready, momentum_swing,
  clutch_moment, injury_scare, persona_revealed, season_start, playoff_push

saveStoryBeat(supabase, beat, { season_id, team_id, player_id }) → row
  Inserts to narrative_events table.

checkStoryTriggers(weekData) → beats[]
  Runs after every weekly tick. Evaluates all 12 trigger conditions.
  Returns array of beats to save.

5.4 draftEngine.js
-------------------
generateDraftClass(draftClassId) → void
  Creates 90 prospects with weighted age distribution, random personas,
  scouting blurbs from prospect_blurb_templates. Inserts to Supabase.

revealProspectAttributes(prospectId, weeksAssigned) → updatedFlags
  Week 1: overall_revealed
  Week 2: skills_revealed
  Week 3: potential_revealed
  Week 4: persona_revealed

runCombine(draftClassId) → void
  Reveals physicals for all 90 prospects. Flags surprises (speed>80 + overall<60)
  and disappointments (speed<50 + overall>65). Sets physicals_revealed=true.

draftPlayer(prospectId, teamId, round, pickNumber) → player
  Converts prospect to player, assigns to dev league, marks pick used.

5.5 chemistryEngine.js
-----------------------
initChemistryPairs(teamId, newPlayerId, newPlayerCategory, existingRoster)
  Creates pair rows for new player vs all existing roster members.
  Initializes at 50% of compatibility ceiling.

updateChemistryPairs(teamId, weekEvents) → void
  Weekly tick: time drift +1, moment spikes, coach modifier.
  Caps at ceiling, floors at 0.
  Updates relationship_type based on score tier.

calcTeamChemistry(teamId) → number (0-100)
  Weighted average of all pair scores by minutes played together.

5.6 legacyEngine.js
--------------------
awardLegacyPoints(gmProfileId, event, points) → newTotal
  Logs to legacy_score_log, calls increment_legacy_score RPC.

checkFranchiseRecords(gmProfileId, seasonData) → brokenRecords[]
  Compares season data against all tracked record types.

evaluateHOFNominees(players) → nominees[]
  Filters players who meet 3+ of 8 HOF criteria.

6. THE WEEKLY TICK — FULL IMPLEMENTATION
================================================================================

  // src/lib/weeklyTick.js
  export async function runWeeklyTick({ teamId, seasonId, weekNumber, supabase }) {

    // 1. Your game already simulated before this runs

    // 2. Sim all 29 AI games
    const { data: allTeams } = await supabase
      .from('teams').select('id, wins, losses').neq('id', teamId)
    const aiResults = simAllAIGames(allTeams)
    await supabase.from('ai_game_results').insert(aiResults)

    // 3. Update all 30 team standings
    for (const r of aiResults) {
      await supabase.from('teams')
        .update({ wins: r.homeScore > r.awayScore ? ... })
        .eq('id', r.winner_id)
    }

    // 4. Dev ticks for your roster
    const { data: roster } = await supabase
      .from('rosters').select('*, players(*)').eq('team_id', teamId)
    const { data: coaches } = await supabase
      .from('coaching_staff').select('*').eq('team_id', teamId)
    for (const entry of roster) {
      const result = growthRoll(entry.players, entry, coaches)
      if (result)
        await supabase.from('players')
          .update({ [result.attribute]: entry.players[result.attribute] + result.delta })
          .eq('id', entry.player_id)
    }

    // 5. Chemistry pairs update
    const weekEvents = await getWeekEvents(weekNumber, seasonId, supabase)
    await updateChemistryPairs(teamId, weekEvents)

    // 6. Recalculate draft pick projected values
    await recalcDraftPickValues(supabase)

    // 7. Check and save story beat triggers
    const weekData = await assembleWeekData(teamId, seasonId, supabase)
    const beats = checkStoryTriggers(weekData)
    for (const beat of beats)
      await saveStoryBeat(supabase, beat, { season_id: seasonId, team_id: teamId })

    // 8. Advance season week
    await supabase.rpc('advance_season_week', { season_id: seasonId })

    // 9. Return new state
    return { success: true, weekNumber: weekNumber + 1 }
  }

  const { data, error } = await supabase.from('players').select('*').eq('id', id)
